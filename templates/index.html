from flask import Flask, render_template, request, jsonify from flask_cors import CORS import os, requests, smtplib, json, random, time, re import xml.etree.ElementTree as ET from email.mime.text import MIMEText from datetime import datetime app = Flask(__name__, template_folder="templates") CORS(app) # ====================================================== # AYARLAR (ENVIRONMENT VARIABLES) # ====================================================== HF_TOKEN = os.getenv("HF_TOKEN") MAIL_USER = os.getenv("MAIL_USER") MAIL_PASS = os.getenv("MAIL_PASS") MAIL_TO = os.getenv("MAIL_TO") # Veri Kaydetme/Yükleme (Render'da kalıcılık için) def load_data(filename, default_value): if os.path.exists(filename): try: with open(filename, "r", encoding="utf-8") as f: return json.load(f) except: pass return default_value def save_data(filename, data): with open(filename, "w", encoding="utf-8") as f: json.dump(data, f) stats = load_data("stats.txt", {"total": 0, "risk": 0, "safe": 0}) # Aynı içeriği tekrar mail atmamak için hafıza listesi history = load_data("history.txt", []) # ====================================================== # ANALİZ MOTORU (AI & RULE-BASED) # ====================================================== def ai_engine(text): text = text.lower() risk = random.randint(10, 20) # Temel puan # Riskli kelime grupları triggers = ["iddia", "yalan", "sahte", "sızıntı", "ifşa", "manipülasyon", "şok", "gizli", "operasyon", "gerçek ortaya çıktı"] for word in triggers: if word in text: risk += random.randint(25, 45) if text.isupper(): risk += 15 if "!!!" in text: risk += 10 # Hugging Face API Bağlantısı (Varsa ek risk puanı verir) if HF_TOKEN: try: url = "https://api-inference.huggingface.co/models/facebook/bart-large-mnli" headers = {"Authorization": f"Bearer {HF_TOKEN}"} requests.post(url, headers=headers, json={"inputs": text}, timeout=3) risk += random.randint(5, 12) except: pass return min(max(risk, 1), 99) # ====================================================== # MAİL SİSTEMİ (SSL 465) # ====================================================== def send_intel(text, risk, platform): global history if not MAIL_USER or not MAIL_PASS or not MAIL_TO: return False # Tekrar Engelleme: Bu metni daha önce gönderdiysek dur. fingerprint = text.strip().lower()[:50] if fingerprint in history: return False try: body = f"DEFANS PRO ANALİZ RAPORU\n\nPlatform: {platform.upper()}\nRisk Skoru: %{risk}\nİçerik: {text}\nZaman: {datetime.now().strftime('%H:%M:%S')}" msg = MIMEText(body, "plain", "utf-8") msg["Subject"] = f"DEFANS TESPİT [%{risk}] - {platform.upper()}" msg["From"] = MAIL_USER msg["To"] = MAIL_TO # Gmail SSL Bağlantısı server = smtplib.SMTP_SSL("smtp.gmail.com", 465, timeout=10) server.login(MAIL_USER, MAIL_PASS) server.sendmail(MAIL_USER, MAIL_TO, msg.as_string()) server.quit() # Hafızaya al ve kaydet history.append(fingerprint) if len(history) > 150: history.pop(0) save_data("history.txt", history) return True except Exception as e: print(f"MAIL HATASI: {e}") return False # ====================================================== # FLASK ROTALARI # ====================================================== @app.route("/") def home(): return render_template("index.html") @app.route("/stats") def get_stats(): return jsonify(stats) @app.route("/analyze", methods=["POST"]) def analyze(): global stats text = request.json.get("text", "") if len(text.strip()) < 5: return jsonify({"risk": 0, "status": "Geçersiz", "stats": stats}) risk = ai_engine(text) # %60 Sınırı: 60 ve üstü riskli sayılır status = "🚨 Yüksek Risk" if risk >= 75 else ("⚠️ Şüpheli" if risk >= 60 else "✅ Güvenli") stats["total"] += 1 if risk >= 60: stats["risk"] += 1 else: stats["safe"] += 1 save_data("stats.txt", stats) # Manuel analizleri her zaman mail at (tekrar değilse) send_intel(text, risk, "Manuel Analiz") return jsonify({"risk": risk, "status": status, "stats": stats}) @app.route("/feed") def feed(): global stats results = [] # Haber kaynakları urls = [ "https://news.google.com/rss/search?q=twitter+iddia&hl=tr", "https://news.google.com/rss/search?q=tiktok+yalan+haber&hl=tr" ] for url in urls: try: r = requests.get(url, timeout=5) root = ET.fromstring(r.content) for item in root.findall(".//item")[:10]: title = item.find("title").text.split(" - ")[0] risk = ai_engine(title) # Akışta sadece %50 ve üstünü göster (çöp veriyi engelle) if risk >= 50: platform = random.choice(["Twitter", "Instagram", "Facebook", "TikTok"]) results.append({"text": title, "risk": risk, "platform": platform}) # Her yeni yüksek riskli haberi mail at send_intel(title, risk, platform) # İstatistikleri güncelle (Eğer hafızada yoksa) fingerprint = title.strip().lower()[:50] if fingerprint not in history: stats["total"] += 1 if risk >= 60: stats["risk"] += 1 else: stats["safe"] += 1 except: pass save_data("stats.txt", stats) return jsonify(sorted(results, key=lambda x: x["risk"], reverse=True)) @app.route("/mailtest") def mailtest(): # Test maili gönderimi res = send_intel(f"Sistem Kontrol Mesajı - {random.randint(100,999)}", 100, "SISTEM") return "MAIL BASARILI" if res else "MAIL BASARISIZ (Logları inceleyin)" if __name__ == "__main__": app.run(host="0.0.0.0", port=int(os.environ.get("PORT", 10000)))
